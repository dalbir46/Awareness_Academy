    // Helper: detect if text needs display-wrapping (common cause of misplaced \hline)

    function needsDisplayWrap(tex){

      const check = /\\begin\{array\}|\\hline|\\begin\{matrix\}|\\begin\{pmatrix\}|\\begin\{pmatrix\}|\\begin\{bmatrix\}/;

      return check.test(tex);

    }

    function wrapIfNeededForRender(tex){

      // If already has display delimiters or $$, return as-is

      if(/^\s*\\\[[\s\S]*\\\]\s*$/.test(tex) || /^\s*\$\$[\s\S]*\$\$\s*$/.test(tex) || /^\s*\\\([\s\S]*\\\)\s*$/.test(tex)) return tex;

      if(needsDisplayWrap(tex)){

        return "\\[" + tex + "\\]";

      }

      return tex;

    }
